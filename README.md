/**
 * Clean & secure JSON response utility
 * @param mixed $data Response data
 * @param int $status HTTP status code
 * @return void
 */
function respond_json($data, int $status = 200): void 
{
    // Set proper HTTP status
    http_response_code($status);
    
    // Security & content headers
    header('Content-Type: application/json; charset=UTF-8');
    header('X-Content-Type-Options: nosniff');
    header('X-Frame-Options: DENY');
    
    // Encode with proper flags
    echo json_encode(
        $data, 
        JSON_UNESCAPED_UNICODE | 
        JSON_UNESCAPED_SLASHES | 
        JSON_THROW_ON_ERROR
    );
    
    exit;
}

// Usage Example
try {
    $result = perform_database_operation();
    
    respond_json([
        'success'   => true,
        'data'      => $result,
        'message'   => 'Operation completed successfully',
        'timestamp' => time(),
        'version'   => '1.0.0'
    ]);
} catch (Exception $e) {
    respond_json([
        'success' => false,
        'error'   => $e->getMessage()
    ], 400);
}

        },
        'args' => [
            'title' => [
                'required'          => true,
                'sanitize_callback' => 'sanitize_text_field',
                'validate_callback' => function($param) {
                    return !empty(trim($param));
                }
            ]
        ]
    ]);

/**
 * GET callback
 */
function hg_get_data(WP_REST_Request $request): WP_REST_Response 
{
    $limit = $request->get_param('limit');
    
    $posts = get_posts([
        'posts_per_page' => $limit,
        'post_status'    => 'publish',
        'orderby'        => 'date',
        'order'          => 'DESC'
    ]);
    
    $data = array_map(function($post) {
        return [
            'id'      => $post->ID,
            'title'   => $post->post_title,
            'excerpt' => wp_trim_words($post->post_content, 20),
            'date'    => $post->post_date
        ];
    }, $posts);
    
    return new WP_REST_Response([
        'success' => true,
        'data'    => $data,
        'count'   => count($data),
        'version' => '1.0.0'
    ], 200);
}

/**
 * POST callback
 */
function hg_create_data(WP_REST_Request $request): WP_REST_Response 
{
    $title = $request->get_param('title');
    
    $post_id = wp_insert_post([
        'post_title'  => $title,
        'post_status' => 'draft',
        'post_author' => get_current_user_id()
    ]);
    
    if (is_wp_error($post_id)) {
        return new WP_REST_Response([
            'success' => false,
            'error'   => $post_id->get_error_message()
        ], 400);
    }
    
    return new WP_REST_Response([
        'success' => true,
        'post_id' => $post_id,
        'message' => 'Post created successfully'
    ], 201);
}

/**
 * Optimized WordPress query with object caching
 * Reduces database load and improves response times
 */
function get_optimized_posts(
    string $post_type = 'post', 
    int $limit = 10,
    bool $force_refresh = false
): array {
    
    // Generate unique cache key
    $cache_key = sprintf(
        'optimized_posts_%s_%d',
        $post_type,
        $limit
    );
    
    // Try to get from cache
    if (!$force_refresh) {
        $cached = wp_cache_get($cache_key, 'custom_queries');
        if (false !== $cached) {
            return $cached;
        }
    }
    
    // Query optimization flags
    $args = [
        'post_type'      => $post_type,
        'posts_per_page' => $limit,
        'post_status'    => 'publish',
        
        // Performance optimizations
        'fields'         => 'ids',           // Only get IDs
        'no_found_rows'  => true,            // Skip total count
        'update_post_meta_cache' => false,   // Skip meta cache
        'update_post_term_cache' => false,   // Skip term cache
        
        'orderby'        => 'date',
        'order'          => 'DESC'
    ];
    
    // Execute optimized query
    $posts = get_posts($args);
    
    // Cache for 1 hour
    wp_cache_set(
        $cache_key, 
        $posts, 
        'custom_queries', 
        HOUR_IN_SECONDS
    );
    
    return $posts;
}

/**
 * Invalidate cache on post update
 */
add_action('save_post', function($post_id, $post) {
    $cache_key = sprintf(
        'optimized_posts_%s_*',
        $post->post_type
    );
    
    wp_cache_delete($cache_key, 'custom_queries');
}, 10, 2);

/**
 * Usage example
 */
function display_recent_posts() {
    $post_ids = get_optimized_posts('post', 5);
    
    foreach ($post_ids as $post_id) {
        // Only load full post object when needed
        $post = get_post($post_id);
        echo '<h3>' . esc_html($post->post_title) . '</h3>';
    }
}
